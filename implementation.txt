6️⃣ PHASE 3 — MOBILE NET SERVICE

Requirements:

Load model only once

Never reload per request

Use CPU

Use torch.no_grad

Return class name, not index

Maintain class label mapping

Example:

import torch
from torchvision import models, transforms
from PIL import Image
import io
from config import MOBILENET_PATH

device = torch.device("cpu")

CLASS_NAMES = [
    "Tomato_Late_blight",
    ...
]

model = models.mobilenet_v3_large()
model.classifier[3] = torch.nn.Linear(1280, len(CLASS_NAMES))
model.load_state_dict(torch.load(MOBILENET_PATH, map_location=device))
model.eval()

transform = transforms.Compose([
    transforms.Resize((224, 224)),
    transforms.ToTensor(),
    transforms.Normalize([0.485,0.456,0.406],
                         [0.229,0.224,0.225])
])

def classify_image(file):
    image_bytes = file.file.read()
    image = Image.open(io.BytesIO(image_bytes)).convert("RGB")
    input_tensor = transform(image).unsqueeze(0)

    with torch.no_grad():
        output = model(input_tensor)
        probs = torch.softmax(output, dim=1)
        confidence, predicted = torch.max(probs, 1)

    disease_name = CLASS_NAMES[predicted.item()]
    confidence_score = float(confidence.item() * 100)

    return disease_name, confidence_score

7️⃣ PHASE 4 — GEMMA LLM SERVICE

Must:

from llama_cpp import Llama
from config import GEMMA_PATH, N_CTX, N_THREADS
import json
import re

llm = Llama(
    model_path=str(GEMMA_PATH),
    n_ctx=N_CTX,
    n_threads=N_THREADS,
    chat_format="gemma",
    verbose=False
)

def extract_json(text):
    text = re.sub(r"```json|```", "", text).strip()
    match = re.search(r"\{.*\}", text, re.DOTALL)
    if match:
        return json.loads(match.group(0))
    raise ValueError("Invalid JSON output")

def generate_disease_report(disease_name, confidence):

    prompt = f"""
    Provide plant disease report in JSON:

    {{
      "diagnosis": "",
      "severity": "",
      "treatment": [],
      "prevention": [],
      "recovery_time": ""
    }}

    Disease: {disease_name}
    """

    response = llm.create_chat_completion(
        messages=[{"role":"user","content":prompt}],
        temperature=0.1,
        max_tokens=400
    )

    raw = response["choices"][0]["message"]["content"]
    parsed = extract_json(raw)

    return {
        "disease_name": disease_name,
        "confidence_score": confidence,
        **parsed
    }

from llm_service import llm, extract_json

def recommend_crop(data):

    prompt = f"""
    Recommend best crop in JSON:

    {{
      "recommended_crop": "",
      "reason": "",
      "care_tips": []
    }}

    Soil Data:
    N: {data.N}
    P: {data.P}
    K: {data.K}
    Temperature: {data.temperature}
    Humidity: {data.humidity}
    pH: {data.pH}
    Rainfall: {data.rainfall}
    """

    response = llm.create_chat_completion(
        messages=[{"role":"user","content":prompt}],
        temperature=0.2,
        max_tokens=300
    )

    return extract_json(response["choices"][0]["message"]["content"])

from fastapi import FastAPI, UploadFile, File
from fastapi.middleware.cors import CORSMiddleware
from mobilenet_service import classify_image
from llm_service import generate_disease_report
from crop_service import recommend_crop
from schemas import CropInput

app.add_middleware(
    CORSMiddleware,
    allow_origins=["*"],
    allow_credentials=True,
    allow_methods=["*"],
    allow_headers=["*"],
)

@app.post("/analyze")
async def analyze(file: UploadFile = File(...)):
    disease, confidence = classify_image(file)
    return generate_disease_report(disease, confidence)

@app.post("/recommend")
async def recommend(data: CropInput):
    return recommend_crop(data)
